function inBox(box) {
	const y = box.findIndex(b => b.includes("*"))
	const x = box[y]?.indexOf("*")

	const positions = [
		box[y - 1]?.[x],
		box[y]?.[x + 1],
		box[y + 1]?.[x],
		box[y]?.[x - 1]
	]

	return positions.every(pos => pos)
}
