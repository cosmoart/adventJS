function organizeShoes(shoes) {
	const counts = {};

	for (const { type, size } of shoes) {
		counts[size] = counts[size] || { I: 0, R: 0 };
		counts[size][type]++;
	}

	return Object.entries(counts)
		.flatMap(([size, { I, R }]) => Array(Math.min(I, R)).fill(+size));
}
