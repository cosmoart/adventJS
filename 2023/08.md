function organizeGifts (gifts) {
	return gifts.match(/\d+[a-z]/gi).reduce((acc, item) => {
		for (let i = item.slice(0, -1); i > 0;) {
			if (i >= 50) {
				acc += `[${item.slice(-1)}]`
				i -= 50
				continue
			}
			if (i >= 10) {
				acc += `{${item.slice(-1)}}`
				i -= 10
				continue
			}
			acc += `(${item.slice(-1).repeat(i)})`
			break
		}
		return acc
	}, "")
}
