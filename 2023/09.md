```js
function adjustLights (lights) {
	let red = 0
	let green = 0
	for (const i in lights) {
		lights[i] !== (i % 2 ? "游릭" : "游댮") && red++
		lights[i] !== (i % 2 ? "游댮" : "游릭") && green++
	}
	return Math.min(red, green)
}
```
