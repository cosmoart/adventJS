function maxGifts(houses) {
  let prevMax = 0;
  let currMax = 0;

  for (let gift of houses) {
    let newMax = Math.max(currMax, prevMax + gift);
    prevMax = currMax;
    currMax = newMax;
  }

  return currMax;
}
