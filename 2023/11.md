function getIndexsForPalindrome (word) {
	if (word === word.split('').reverse().join('')) return []

	for (const i in word) {
		for (let j = +i + 1; j < word.length; j++) {
			const arr = word.split("")
			const temp = arr[i];
			arr[i] = arr[j];
			arr[j] = temp;

			if (arr.join("") === arr.reverse().join("")) return [+i, j]
		}
	}

	return null
}
