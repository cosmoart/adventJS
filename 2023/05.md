```js
function cyberReindeer(road, time) {
	let pos = 0
	return new Array(time).fill(road.replaceAll("S", "."))
  .reduce((acc, char, i) => {
		const para = char.slice(0, pos) + "S" + char.slice(pos + 1)
		if (road[pos + 1] === "." || i >= 4) pos++
		return [...acc, para.replaceAll("|", i >= 5 ? "*" : "|")]
	}, [])
}
```
